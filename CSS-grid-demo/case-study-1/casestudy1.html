  /* Repeat a 1fr column 3 times */
  /* Two rows: 100px and 200px */
  /* positioning the elements in the markup using shorthand syntax declaring the start and end values at once */
  /* if the end value is spanning one track then it would be valid to omit the / and end value */
  .grid-container {
    grid-template-columns: repeat(3, 1fr);
    grid-template-rows: 100px 200px;
    grid-gap: 20px
  }

  .item-1 {
    grid-column: 2 / 3; /* same as grid-column: 2 */
    grid-row: 1 / 2;    /* same as grid-row: 1 */ 
  }
  .item-2 {
    /* not able to omit the end vaule as it is spanning more than one track */
    grid-column: 2 / 4; 
    grid-row: 2 / 3;    /* same as grid-row: 2 */
  }
  .item-3 {
    grid-column: 3 / 4; /* same as grid-column: 3 */
    grid-row: 1 / 2;    /* same as grid-row: 1 */
  }
  .item-4 {
    grid-column: 1 / 2; /* same as grid-column: 1 */
    grid-row: 2 / 3;    /* same as grid-row: 2 */
  }
  .item-5 {
    grid-column: 1 / 2; /* same as grid-column: 1 */
    grid-row: 1 / 2;    /* same as grid-row: 1 */
  }
  

  
  /* Repeat a 1fr column 3 times */
  /* Three rows: 1fr 200px 1fr */
  /* using the span keyword rather than targeting the Grid Line by number */
  /* also using the defaults for row and column end, which is to span 1 */

  .grid-container {
    grid-template-columns: repeat(3, 1fr);
    grid-template-rows: 1fr 200px 1fr;
    grid-gap: 20px
  }

  .item-1 {
    grid-column: 2 / span 2;
    grid-row: 1;
  }
  .item-2 {
    grid-column: 1;
    grid-row: 1 / span 2;
  }
  .item-3 {
    grid-column: 3;
    grid-row: 2;
  }
  .item-4 {
    grid-column: 1 / span 3;
    grid-row: 3;
  }
  .item-5 {
    grid-column: 2;
    grid-row: 2;
  }
  

  
  /* 4 rows : repeat a 1fr row 4 times
     3 columns : repeat a 1fr column 3 times */

     and then using the grid-template-areas property to describe
